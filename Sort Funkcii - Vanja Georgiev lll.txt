#include <bits/stdc++.h>
using namespace std;

void insertion_sort (int a[], int n)
{
    for(int r=1;r<n;r++)
    {
        int t=a[r];
        for(int i=r-1;i>=0;i--)
        {
            if(a[i]>t)
            {
                a[i+1]=a[i];
                a[i]=t;
            }
            else if(a[i]<t)
            {
                a[i+1]=t;
                break;
            }
        }
    }
}

{

    for(int i=0;i<n;i++)
    {
        a[i]=dis(mt);
    }
    insertion_sort(a,n);
        delete[] a;

}

void selection_sort (int a[], int n)
{
    for (int i=0;i<n-1;i++)
    {
        int m=a[i],t=a[i],x;
        for(int j=i+1;j<n;j++)
        {
            if(a[j]<m)
            {
                m=a[j];
                x=j;
            }
        }
        if (m!=a[i])
        {
            a[i]=m;
            a[x]=t;
        }
    }
}

#include <bits/stdc++.h>
using namespace std;

void merge_(int a[],int l, int m, int r)
{
    int n1 = m - l + 1;
    int n2 = r - m;
    int* L = new int[n1];
    int* R = new int[n2];
    for (int i = 0; i < n1; i++) {
        L[i] = a[l + i];
    }
    for (int i = 0; i < n2; i++) {
        R[i] = a[m + 1 + i];
    }
    int i = 0, j = 0, k = 1;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) {
            a[k++] = L[i++];
        }
        else {
            a[k++] = R[j++];
        }
    }
    while (i < n1) {
        a[k++] = L[i++];
    }
    while (j < n2) {
        a[k++] = R[j++];
    }
    delete[] L;
    delete[] R;
}

void merge_sort(int a[], int l, int r)
{
    if (l<r) {
        int m = l+(r-1)/2;
        merge_sort(a, l, m);
        merge_(a,l,m,r);
    }
}

#include <bits/stdc++.h>
using namespace std;

void quick_sort (int a[], int n)
{
    int p=a[n-1],t,i=-1;
    while (i<n)
    {
        for(int j=0;j<n;j++)
        {
            if(a[j]<p)
            {
                i++;
                t=a[i];
                a[i]=a[j];
                a[j]=t;
            }
        }
        i++;
        t=a[i];
        a[i]=p;
        a[n-1]=t;
        break;
    }
    int b[i],c[n-i-1];
    int m=i,q=n-i-1;
    for (int j=0;j<m;j++)
        b[j]=a[j];
    for (int j=0;j<q;j++)
        c[j]=a[j+i+1];
    if (m>1)
        quick_sort(b,m);
    if (q>1)
        quick_sort(c,q);
    for (int j=0;j<m;j++)
        a[j]=b[j];
    for (int j=0;j<q;j++)
        a[j+i+1]=c[j];
}

{
    const int n=100000;
